class Poll {
    constructor({ task, wait, success, fail, limitTimes }) {
        this.task = _ => new Promise(task)
        this.wait = wait
        this.limitTimes = limitTimes
        this.timer = ''
        this.success = success
        this.fail = fail
        this.currentTimes = 0
        this.run()
    }

    async run() {
        this.currentTimes++
        const res = await this.task()
        if (res) {
            this.success(res)

            if (this.currentTimes === this.limitTimes) { //è‹¥æœ€åä¸€æ¬¡è½®è¯¢ è¿˜æ˜¯å¤±è´¥ åˆ™æ‰§è¡Œå¤±è´¥å›è°ƒ

    }
